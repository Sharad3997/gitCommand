Git command:
//check git version
git --version

//for help git purpose

git help

//to check any git initialize

git status

// to git initialize

git init

//create file

touch file name

//file add to git

git add file_name  (for a single file add to git)
git add .          (for all file or folder which you entered in git)

//to initialize user name

git config --global user.name

//to initialize email 

git config --global user.email

//link with remote server to git file

git remote add origin repo_name

//file pust to the remote server

git push origin master

//clone of git folder to remote server

git clone reo_name

//pull file from remote server

git  pull origin master
